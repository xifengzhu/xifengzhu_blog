require 'mechanize'
class Douban
  def initialize
    @agent = Mechanize.new
    @page = @agent.get('http://www.doubanmeizi.com')
  end

  # loop the paginate
  def loop_page(page = @page)
    snatch_images(page)
    next_page_link = page.links.find{|link| link.text.eql?("ä¸‹ä¸€é¡µ")}
    page = next_page_link.nil? ? nil : (@agent.click next_page_link)
    loop_page(page) unless page.nil?
  end

  # click all the detail href and snatch the origin picture
  def snatch_images(page)
    puts "start save image...."
    Dir.chdir("./image") do |path|
      page.search('a.imageLink').each do |link|
        new_page = @agent.click link
        new_page.search("img.alignnone").each do |image|
          image_url = image.attr("src")
          puts image_url
          @agent.get(image_url) do |image|
            image.save
          end
        end
      end
    end
  end

  def start
    loop_page()
  end
end
